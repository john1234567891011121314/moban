## __int128
```c++
// int128
using i128 = __int128;
std::array<char, 110> u{};
std::ostream& operator<<(std::ostream& os, __int128 v)
{
    if (v < 0)
    {
        v = -v;
        os << '-';
    }
    char* i = u.data() + 60;
    *i = '\0';
    do *--i = v % 10 + '0', v /= 10;
    while (v > 0);
    return cout << i;
}
i128 toi128(const std::string& s)
{
    i128 n = 0;
    int f = 1;
    for (auto c : s)
    {
        if (c == '-')
            f = -1;
        else
            n = n * 10 + (c - '0');
    }
    return n * f;
}
i128 sqrti128(i128 n)
{
    i128 lo = 0, hi = 1E16;
    while (lo < hi)
    {
        i128 x = (lo + hi + 1) / 2;
        if (x * x <= n)
        {
            lo = x;
        }
        else
        {
            hi = x - 1;
        }
    }
    return lo;
}

i128 gcd(i128 a, i128 b)
{
    while (b)
    {
        a %= b;
        std::swap(a, b);
    }
    return a;
}
```
