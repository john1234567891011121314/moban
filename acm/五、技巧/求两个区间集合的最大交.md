## 求两个区间集合的最大交
```c++
	sort(a.begin(), a.end());
	sort(b.begin(), b.end());
	int n1 = a.size(), n2 = b.size();
	ll ans = 0;
	for(int i = 0, j = 0; i < n1 && j < n2;)
	{
		ll l1 = a[i].first, r1 = a[i].second;
		ll l2 = b[j].first, r2 = b[j].second;
		if(r2 >= l1 && r1 >= l2)
		{
			ans = max(ans, min(r1, r2) - max(l1, l2));
		}
		if(r2 < r1) j++;
		else i++;
	}
	cout << ans << '\n';
```
