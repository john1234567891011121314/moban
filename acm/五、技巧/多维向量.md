## 多维向量
```c++
template<typename T>
T ndvector(T v){
    return v;
}

template<typename T,typename...Args>
auto ndvector(T v, int n, Args...args)
{
    return std::vector<decltype(ndvector(v, args...))> (n, ndvector(v, args...));
}

//auto vec = ndvector("#", 4, 5); 创建一个4 * 5 的初始化为"#"的向量


template<typename T>
std::ostream& operator<<(std::ostream& os, const std::vector<T>& vec)
{
    os << "[ " << vec.size() << " : { ";
    for(const auto& element : vec)
    {
        os << element << ' '; 
    }
    os << "} ]";
    return os;
}
```
