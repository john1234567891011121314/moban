## 区间合并
```c++
 vector<vector<int>> merge(vector<vector<int>>& intervals) {
        sort(intervals.begin(), intervals.end());
        vector<vector<int>>ans;
        int st = INT_MIN, ed = INT_MIN;
        for(auto v:intervals)
        {
            if(ed == INT_MIN)
            {
                st = v[0];
                ed = v[1];
            }
            else if(v[0] <= ed)
            {
                ed = max(v[1], ed);
            }
            else if(v[0] > ed)
            {
                ans.push_back({st, ed});
                st = v[0];
                ed = v[1];
            }
        }
        ans.push_back({st, ed});
        return ans;
}
```
