template <class T>
[[nodiscard]] inline constexpr T qpow(T base, ll power) noexcept
{
    T res{1};
    while (power)
    {
        if (power & 1) res = res * base;
        base = base * base;
        power >>= 1;
    }
    return res;
}

struct mint
{
    int x;

    [[nodiscard]] static constexpr int norm(int x) noexcept
    {
        x += (x >> 31) & mod;
        x -= (x >= mod) ? mod : 0;
        return x;
    }

    constexpr mint() noexcept : x(0) {}
    constexpr mint(int v) noexcept : x(norm(v)) {}
    constexpr mint(long long x) noexcept : x(norm(static_cast<int>(x % mod))) {}
    [[nodiscard]] constexpr int val() const noexcept { return x; }
    [[nodiscard]] constexpr mint operator-() const noexcept { return mint(mod - x); }
    [[nodiscard]] mint inv() const noexcept
    {
        if (__builtin_expect(x == 0, 0)) [[unlikely]] {
            assert(false);          
            __builtin_unreachable(); 
        }
        return qpow(*this, mod - 2);
    }
    
    constexpr mint &operator+=(const mint &rhs) noexcept
    {
        x += rhs.x;
        x -= (x >= mod) ? mod : 0;
        return *this;
    }
    constexpr mint &operator-=(const mint &rhs) noexcept
    {
        x -= rhs.x;
        x += (x >> 31) & mod;
        return *this;
    }
    constexpr mint &operator*=(const mint &rhs) noexcept
    {
        x = static_cast<int>((long long)x * rhs.x % mod);
        return *this;
    }
    constexpr mint &operator/=(const mint &rhs) noexcept { return *this *= rhs.inv(); }
    [[nodiscard]] friend constexpr bool operator==(const mint &lhs, const mint &rhs) noexcept
    {
        return lhs.x == rhs.x;
    }
    [[nodiscard]] friend constexpr mint operator+(mint lhs, const mint &rhs) noexcept
    {
        lhs += rhs;
        return lhs;
    }
    [[nodiscard]] friend constexpr mint operator-(mint lhs, const mint &rhs) noexcept {
        lhs -= rhs;
        return lhs;
    }
    [[nodiscard]] friend constexpr mint operator*(mint lhs, const mint &rhs) noexcept
    {
        lhs *= rhs;
        return lhs;
    }

    [[nodiscard]] friend constexpr mint operator/(mint lhs, const mint &rhs) noexcept {
        lhs /= rhs;
        return lhs;
    }
    friend std::istream &operator>>(std::istream &is, mint &a)
    {
        ll v;
        is >> v;
        v %= mod;
        a.x = static_cast<int>(v < 0 ? v + mod : v);
        return is;
    }
    friend std::ostream &operator<<(std::ostream &os, const mint &a)
    {
        return os << a.x;
    }
};

template <std::uint32_t P>struct MontInt {
	using u32 = std::uint32_t;
	using u64 = std::uint64_t;
	u32 v;
	static constexpr u32 get_r() {
		u32 iv = P;
 
        for (u32 i = 0; i != 4; ++i)
            iv *= 2U - P * iv;
 
        return -iv;
	}
    static constexpr u32 r = get_r(), r2 = -u64(P) % P;//限定词不能省 
    
	MontInt() = default;
    ~MontInt() = default;
    constexpr MontInt(u32 v) : v(reduce(u64(v) * r2)) {}
    constexpr MontInt(const MontInt &rhs) : v(rhs.v) {}
    
    static constexpr u32 reduce(u64 x) {
        return x + (u64(u32(x) * r) * P) >> 32;
    }
    
    static constexpr u32 norm(u32 x) {
        return x - (P & -(x >= P));
    }
    constexpr u32 get() const{
        u32 res = reduce(v) - P;
        return res + (P & -(res >> 31));
    }
    
    
    constexpr MontInt operator-() const {
        MontInt res;
        return res.v = (P << 1 & -(v != 0)) - v, res;
    }
    constexpr MontInt inv() const {
        return pow(-1);
    }
    constexpr MontInt &operator=(const MontInt &rhs) {
        return v = rhs.v, *this;
    }
    constexpr MontInt &operator+=(const MontInt &rhs) {
        return v += rhs.v - (P << 1), v += P << 1 & -(v >> 31), *this;
    }
    constexpr MontInt &operator-=(const MontInt &rhs) {
        return v -= rhs.v, v += P << 1 & -(v >> 31), *this;
    }
    constexpr MontInt &operator*=(const MontInt &rhs) {
        return v = reduce(u64(v) * rhs.v), *this;
    }
    constexpr MontInt &operator/=(const MontInt &rhs) {
        return this->operator*=(rhs.inv());
    }
    friend MontInt operator+(const MontInt &lhs,
                                        const MontInt &rhs) {
        return MontInt(lhs) += rhs;
    }
    friend MontInt operator-(const MontInt &lhs,
                                        const MontInt &rhs) {
        return MontInt(lhs) -= rhs;
    }
    friend MontInt operator*(const MontInt &lhs,
                                        const MontInt &rhs) {
        return MontInt(lhs) *= rhs;
    }
    friend MontInt operator/(const MontInt &lhs,
                                        const MontInt &rhs) {
        return MontInt(lhs) /= rhs;
    }
    friend bool operator==(const MontInt &lhs, const MontInt &rhs) {
        return norm(lhs.v) == norm(rhs.v);
    }
    friend bool operator!=(const MontInt &lhs, const MontInt &rhs) {
        return norm(lhs.v) != norm(rhs.v);
    }
    constexpr MontInt pow(ll y) const {
        if ((y %= P - 1) < 0)
            y += P - 1; // phi(P) = P - 1, assume P is a prime number
 
        MontInt res(1), x(*this);
 
        for (; y != 0; y >>= 1, x *= x)
            if (y & 1)
                res *= x;
 
        return res;
    }
};
